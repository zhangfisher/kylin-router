```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Kylin Router - 组件加载系统演示</title>

</head>
<body>
    <div id="app">
        <nav class="demo-nav">
            <h1>组件加载系统演示</h1>
            <kylin-link href="/loader/local">本地组件</kylin-link>
            <kylin-link href="/loader/dynamic">动态导入</kylin-link>
            <kylin-link href="/loader/remote">远程 HTML</kylin-link>
            <kylin-link href="/loader/nested">嵌套路由</kylin-link>
            <kylin-link href="/loader/data">数据预加载</kylin-link>
        </nav>
        <main>
            <kylin-outlet path="/loader"></kylin-outlet>
        </main>
        <div class="demo-log">
            <h3>导航日志</h3>
            <div id="log-container"></div>
        </div>
    </div>

    <script type="module">
        import { KylinRouter } from '/src/index.ts';
        
        const host = document.getElementById('app');
        
        const router = new KylinRouter(host, {
            routes: [
                // 本地组件示例
                {
                    path: '/loader/local',
                    component: 'div',
                    renderEach: (to, from, next) => {
                        setTimeout(() => {
                            next({ title: '本地组件示例', content: '这是一个本地 HTML 元素' });
                        }, 500);
                    }
                },
                // 动态导入示例
                {
                    path: '/loader/dynamic',
                    component: async () => {
                        // 模拟动态导入
                        return new Promise(resolve => {
                            setTimeout(() => {
                                resolve('div');
                            }, 1000);
                        });
                    }
                },
                // 远程 HTML 示例
                {
                    path: '/loader/remote',
                    component: 'https://jsonplaceholder.typicode.com/',  // 示例 URL
                    remoteOptions: {
                        timeout: 5000
                    }
                },
                // 嵌套路由示例
                {
                    path: '/loader/nested',
                    component: 'div',
                    children: [
                        {
                            path: '/loader/nested/child',
                            component: 'span'
                        }
                    ]
                },
                // 数据预加载示例
                {
                    path: '/loader/data',
                    component: 'div',
                    renderEach: async (to, from, next) => {
                        // 模拟 API 调用
                        const response = await fetch('https://jsonplaceholder.typicode.com/posts/1');
                        const data = await response.json();
                        next({ postData: data });
                    }
                }
            ]
        });
        
        router.attach();
        
        // 日志记录
        router.addEventListener('route-change', (e) => {
            addLog(`路由变化: ${e.detail.route.path}`);
        });
        
        function addLog(message) {
            const container = document.getElementById('log-container');
            const log = document.createElement('div');
            log.textContent = `${new Date().toLocaleTimeString()} - ${message}`;
            container.appendChild(log);
        }
    </script>
</body>
</html>
```

```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Kylin Router - 错误处理和重试机制演示</title>
    <link rel="stylesheet" href="../styles.css">
</head>
<body>
    <div id="app">
        <nav class="demo-nav">
            <h1>错误处理和重试机制演示</h1>
            <kylin-link href="/error/404">404 错误</kylin-link>
            <kylin-link href="/error/load-fail">加载失败</kylin-link>
            <kylin-link href="/error/timeout">超时错误</kylin-link>
            <kylin-link href="/error/retry">自动重试</kylin-link>
        </nav>
        <main>
            <kylin-outlet path="/error"></kylin-outlet>
        </main>
        <div class="error-log">
            <h3>错误日志</h3>
            <div id="error-container"></div>
        </div>
    </div>

    <script type="module">
        import { KylinRouter } from '/src/index.ts';
        
        const host = document.getElementById('app');
        
        // 自定义错误组件
        const ErrorComponent = (props) => {
            return `
                <div class="error-boundary">
                    <h2>出错了！</h2>
                    <p>${props.error.message}</p>
                    <button onclick="router.retryLoad()">重试</button>
                    <kylin-link href="/">返回首页</kylin-link>
                </div>
            `;
        };
        
        const router = new KylinRouter(host, {
            routes: [
                // 404 错误示例
                {
                    path: '/error/404',
                    component: 'div',
                    errorBoundary: {
                        component: ErrorComponent,
                        fallback: '<div>页面不存在</div>'
                    }
                },
                // 加载失败示例
                {
                    path: '/error/load-fail',
                    component: 'invalid-url-that-will-fail',
                    errorBoundary: {
                        component: ErrorComponent,
                        retry: { max: 3, delay: 1000 }
                    }
                },
                // 超时错误示例
                {
                    path: '/error/timeout',
                    component: async () => {
                        // 模拟超时
                        await new Promise(resolve => setTimeout(resolve, 10000));
                        return 'div';
                    },
                    timeout: 2000,
                    errorBoundary: {
                        component: ErrorComponent
                    }
                },
                // 自动重试示例
                {
                    path: '/error/retry',
                    component: async () => {
                        // 模拟失败后成功
                        const attempts = window.retryAttempts || 0;
                        window.retryAttempts = attempts + 1;
                        
                        if (attempts < 2) {
                            throw new Error(`加载失败（尝试 ${attempts + 1}/3）`);
                        }
                        
                        return 'div';
                    },
                    retry: {
                        max: 3,
                        delay: 1000,
                        onRetry: (attempt, error) => {
                            console.log(`重试 ${attempt}:`, error.message);
                        }
                    },
                    data: {
                        title: '重试成功',
                        content: '经过几次重试后，组件加载成功！'
                    }
                }
            ],
            defaultErrorComponent: ErrorComponent
        });
        
        router.attach();
        
        // 错误事件监听
        router.addEventListener('component-error', (e) => {
            addErrorLog(e.detail.error.message);
        });
        
        function addErrorLog(message) {
            const container = document.getElementById('error-container');
            const log = document.createElement('div');
            log.className = 'error-entry';
            log.textContent = `${new Date().toLocaleTimeString()} - ${message}`;
            container.appendChild(log);
        }
    </script>
</body>
</html>
```
