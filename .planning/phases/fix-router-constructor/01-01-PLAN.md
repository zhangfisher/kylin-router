在构造函数之前添加类型守卫函数，用于区分 KylinRouterOptiopns 和 KylinRoutes：

```typescript
/**
 * 类型守卫：检查是否为完整的路由配置对象
 * 通过检查是否存在 routes 属性来区分 KylinRouterOptiopns 和 KylinRoutes
 */
function isRouteConfig(options: any): options is KylinRouterOptiopns {
    return (
        typeof options === "object" &&
        options !== null &&
        "routes" in options
    );
}

/**
 * 类型守卫：检查是否为路由项或路由项数组
 * 通过检查是否有 path 和 name 属性来识别 RouteItem
 */
function isRouteItem(options: any): options is RouteItem | RouteItem[] {
    if (Array.isArray(options)) {
        return true;
    }
    return (
        typeof options === "object" &&
        options !== null &&
        "path" in options &&
        "name" in options
    );
}
```

关键设计决策：
- 使用 `"routes" in options` 而不是 `"path" in options`，因为只有完整配置对象才有 routes 属性
- 先检查数组，避免对数组进行 `"path" in options` 检查（数组有索引属性）
- 使用 typeof 检查排除函数和字符串类型
