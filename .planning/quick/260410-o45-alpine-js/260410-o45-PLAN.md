**文件:** `src/utils/hash.ts` (新建)

- 处理特殊字符转义，确保生成的 hash 是有效的 Alpine.js store 名称

**内容:**
```typescript
import type { RouteItem } from "@/types/routes";
import { params } from "./params";

/**
 * 生成路由哈希标识
 * @param route - 路由对象
 * @param router - 路由器实例
 * @returns 生成的哈希字符串
 */
export function generateRouteHash(
    route: RouteItem,
    currentPath: string,
    currentParams: Record<string, string>,
    currentQuery: Record<string, string>
): string {
    const hashPattern = route.hash || "{path}";

    // 构建插值变量
    const interpolationVars = {
        path: route.path,
        basepath: route.path,
        url: currentPath,
        timestamp: Date.now(),
        ...currentQuery,
        ...currentParams,
    };

    // 使用插值函数生成 hash
    let hash = params(hashPattern, interpolationVars);

    // 转义特殊字符，确保是有效的 Alpine.js store 名称
    // 替换非字母数字字符为下划线
    hash = hash.replace(/[^a-zA-Z0-9_-]/g, '_');

    return hash;
}
```

@customElement("kylin-outlet")
export class KylinOutletElement extends KylinRouterElementBase {
    static styles = styles;

    @property({ type: String, reflect: true })
    path?: string;

    @property({ type: Boolean, reflect: true })
    isModal: boolean = false;

    @property({ type: Number })
    timeout: number = 5000;

    @property({ type: Boolean })
    cacheable: boolean = true;

    @property({ type: String, reflect: true })
    renderMode?: "replace" | "append";

    /** 新增：启用 keepalive 缓存 */
    @property({ type: Boolean, reflect: true })
    keepalive: boolean = false;

    /** Alpine.js store 名称 */
    private _currentStoreName?: string;

    /** 缓存的视图容器 */
    private _viewCache: Map<string, HTMLElement> = new Map();

    /** 当前激活的视图 hash */
    private _activeHash?: string;

    // ... 其他代码

    /**
     * 添加 Alpine.js store
     * @param name - store 名称
     * @param data - store 数据
     */
    addStore(name: string, data: Record<string, any>): void {
        // 注册 Alpine.data
        if (typeof window !== 'undefined' && (window as any).Alpine) {
            (window as any).Alpine.data(name, () => data);
            this._currentStoreName = name;

            // 设置 x-data 属性
            this.setAttribute('x-data', name);
        }
    }

    /**
     * 渲染视图内容
     * @param content - 要渲染的内容
     * @param hash - 路由哈希
     * @param hasData - 是否有数据
     * @param data - 路由数据
     */
    async renderView(
        content: any,
        hash: string,
        hasData: boolean,
        data?: Record<string, any>
    ): Promise<void> {
        // 如果启用 keepalive，检查缓存
        if (this.keepalive && this._viewCache.has(hash)) {
            const cachedView = this._viewCache.get(hash)!;
            this._switchToView(hash, cachedView);
            return;
        }

        // 创建新的视图容器
        const viewContainer = document.createElement('div');
        viewContainer.setAttribute('data-hash', hash);

        // 根据 hasData 决定是否创建 Alpine.data
        if (hasData && data) {
            // 注册 Alpine.js store
            this.addStore(hash, data);
            viewContainer.setAttribute('x-data', hash);
        }

        // 渲染内容
        if (typeof content === 'string') {
            viewContainer.innerHTML = content;
        } else if (content instanceof HTMLElement) {
            viewContainer.appendChild(content);
        } else if (content?.template) {
            // lit 模板渲染
            // ... 处理 lit 模板
        }

        // 如果启用 keepalive，缓存视图
        if (this.keepalive) {
            this._viewCache.set(hash, viewContainer);
        }

        // 切换到新视图
        this._switchToView(hash, viewContainer);
    }

    /**
     * 切换到指定视图
     * @param hash - 视图哈希
     * @param view - 视图元素
     */
    private _switchToView(hash: string, view: HTMLElement): void {
        // 隐藏当前视图
        if (this._activeHash && this._activeHash !== hash) {
            const currentView = this.querySelector(`[data-hash="${this._activeHash}"]`);
            if (currentView) {
                currentView.style.display = 'none';
            }
        }

        // 显示新视图
        view.style.display = '';

        // 如果视图还未添加到 DOM，添加它
        if (!view.parentElement) {
            this.appendChild(view);
        }

        this._activeHash = hash;

        // 如果未启用 keepalive，清理旧视图
        if (!this.keepalive) {
            this._cleanupInactiveViews(hash);
        }
    }

    /**
     * 清理非活动视图（未启用 keepalive 时）
     * @param activeHash - 当前活动的哈希
     */
    private _cleanupInactiveViews(activeHash: string): void {
        const views = this.querySelectorAll('[data-hash]');
        views.forEach(view => {
            const hash = view.getAttribute('data-hash');
            if (hash !== activeHash) {
                view.remove();
                // 清理对应的 Alpine.data
                if (hash && typeof window !== 'undefined' && (window as any).Alpine) {
                    // Alpine.js 会自动清理，但我们可以主动移除 store
                    delete (window as any).Alpine.store(hash);
                }
            }
        });
    }

    disconnectedCallback() {
        super.disconnectedCallback();

        // 清理所有缓存
        this._viewCache.clear();

        // 清理 Alpine.data
        if (this._currentStoreName && typeof window !== 'undefined' && (window as any).Alpine) {
            // Alpine.js 会自动清理
        }
    }
}

import { generateRouteHash } from "@/utils/hash";

// 在 renderToOutlet 方法中
async renderToOutlet(
    content: any,
    outlet: any,
    options?: RenderOptions
): Promise<void> {
    const route = this.router.routes.current.route;
    const hasData = route?.data && Object.keys(route.data).length > 0;

    // 生成路由哈希
    const hash = generateRouteHash(
        route,
        this.router.history.location.pathname,
        this.router.routes.current.params,
        this.router.routes.current.query
    );

    // 获取路由数据
    const data = hasData ? (route as any).data : undefined;

    // 调用 outlet 的 renderView 方法
    await outlet.renderView(content, hash, hasData, data);
}
